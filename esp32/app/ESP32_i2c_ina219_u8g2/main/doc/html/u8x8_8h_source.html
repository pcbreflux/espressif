36  U8glib has several layers. Each layer is implemented with a callback function.
37  This callback function handels the messages for the layer.
38 
39  The topmost level is the display layer. It includes the following messages:
40 
41  U8X8_MSG_DISPLAY_SETUP_MEMORY no communicaation with the display, setup memory ony
42  U8X8_MSG_DISPLAY_INIT
43  U8X8_MSG_DISPLAY_SET_FLIP_MODE
44  U8X8_MSG_DISPLAY_SET_POWER_SAVE
45  U8X8_MSG_DISPLAY_SET_CONTRAST
46  U8X8_MSG_DISPLAY_DRAW_TILE
47 
48  A display driver may decided to breakdown these messages to a lower level interface or
49  implement this functionality directly.
50 
51 
52  One layer is the Command/Arg/Data interface. It can be used by the display layer
53  to communicate with the display hardware.
54  This layer only deals with data, commands and arguments. D/C line is unknown.
55  U8X8_MSG_CAD_INIT
56  U8X8_MSG_CAD_SET_I2C_ADR (obsolete)
57  U8X8_MSG_CAD_SET_DEVICE (obsolete)
58  U8X8_MSG_CAD_START_TRANSFER
59  U8X8_MSG_CAD_SEND_CMD
60  U8X8_MSG_CAD_SEND_ARG
61  U8X8_MSG_CAD_SEND_DATA
62  U8X8_MSG_CAD_END_TRANSFER
63 
64  The byte interface is there to send 1 byte (8 bits) to the display hardware.
65  This layer depends on the hardware of a microcontroller, if a specific hardware
66  should be used (I2C or SPI).
67  If this interface is implemented via software, it may use the GPIO level for sending
68  bytes.

207 struct u8x8_display_info_struct
208 {
209  /* == general == */
210 
211  uint8_t chip_enable_level; /* UC1601: 0 */
212  uint8_t chip_disable_level; /* opposite of chip_enable_level */
213 
214  uint8_t post_chip_enable_wait_ns; /* UC1601: 5ns */
215  uint8_t pre_chip_disable_wait_ns; /* UC1601: 5ns */
216  uint8_t reset_pulse_width_ms; /* UC1601: 0.003ms --> 1ms */
217  uint8_t post_reset_wait_ms; /* UC1601: 6ms */
218 
219 
220  /* == SPI interface == */
221 
222  /* after SDA has been applied, wait this much time for the SCK data takeover edge */
223  /* if this is smaller than sck_pulse_width_ns, then use the value from sck_pulse_width_ns */
224  uint8_t sda_setup_time_ns; /* UC1601: 12ns */
225  /* the pulse width of the the clock signal, cycle time is twice this value */
226  /* max freq is 1/(2*sck_pulse_width_ns) */
227  /* AVR: below 70: DIV2, 8 MHz, >= 70 --> 4MHz clock (DIV4) */
228  uint8_t sck_pulse_width_ns; /* UC1701: 50ns */
229 
230  /* until here we have 8 bytes (uint8_t). Newly introduced for SPI.beginTransaction */
231  uint32_t sck_clock_hz;
232 
233  /* previous name "sck_takeover_edge" renamed to "spi_mode" */
234  /* bit 0 of spi_mode is equal to the value of the previous variable sck_takeover_edge, 20 Aug 16: This is wrong the bit is actually inverted */
235  /* SPI has four clock modes: */
236  /* 0: clock active high, data out on falling edge, clock default value is zero, takover on rising edge */
237  /* 1: clock active high, data out on rising edge, clock default value is zero, takover on falling edge */
238  /* 2: clock active low, data out on rising edge */
239  /* 3: clock active low, data out on falling edge */
240  /* most displays have clock mode 1 */
241  uint8_t spi_mode;
242 
243  /* == I2C == */
244  uint8_t i2c_bus_clock_100kHz; /* UC1601: 1000000000/275 = 37 *100k */
245 
246 
247  /* == 8 bit interface == */
248 
249  /* how long to wait after all data line are set */
250  uint8_t data_setup_time_ns; /* UC1601: 30ns */
251  /* write enable pulse width */
252  uint8_t write_pulse_width_ns; /* UC1601: 40ns */
253 
254  /* == layout == */
255  uint8_t tile_width;
256  uint8_t tile_height;
257 
258  uint8_t default_x_offset; /* default x offset for the display */
259  uint8_t flipmode_x_offset; /* x offset, if flip mode is enabled */
260 
261  /* pixel width is not used by the u8x8 procedures */
262  /* instead it will be used by the u8g2 procedures, because the pixel dimension can */
263  /* not always be calculated from the tile_width/_height */
264  /* the following conditions must be true: */
265  /* pixel_width <= tile_width*8 */
266  /* pixel_height <= tile_height*8 */
267  uint16_t pixel_width;
268  uint16_t pixel_height;
269 };

419 /*
420  Name: U8X8_MSG_DISPLAY_DRAW_TILE
421  Args:
422  arg_int: How often to repeat this tile pattern
423  arg_ptr: pointer to u8x8_tile_t
424  uint8_t *tile_ptr; pointer to one or more tiles (number is "cnt")
425  uint8_t cnt; number of tiles
426  uint8_t x_pos; first tile x position
427  uint8_t y_pos; first tile y position
428  Tasks:
429  One tile has exactly 8 bytes (8x8 pixel monochrome bitmap).
430  The lowest bit of the first byte is the upper left corner
431  The highest bit of the first byte is the lower left corner
432  The lowest bit of the last byte is the upper right corner
433  The highest bit of the last byte is the lower left corner
434  "tile_ptr" is the address of a memory area, which contains
435  one or more tiles. "cnt" will contain the exact number of
436  tiles in the memory areay. The size of the memory area is 8*cnt;
437  Multiple tiles in the memory area form a horizontal sequence, this
438  means the first tile is drawn at x_pos/y_pos, the second tile is drawn
439  at x_pos+1/y_pos, third at x_pos+2/y_pos.
440  "arg_int" tells how often the tile sequence should be repeated:
441  For example if "cnt" is two and tile_ptr points to tiles A and B,
442  then for arg_int = 3, the following tile sequence will be drawn:
443  ABABAB. Totally, cnt*arg_int tiles will be drawn.
444 
445 */
446 #define U8X8_MSG_DISPLAY_DRAW_TILE 15
